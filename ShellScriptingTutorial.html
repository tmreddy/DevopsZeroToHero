        ## provide meta data
        # explain steps divisible by 3, divisible by 5, not divisible by 15 range is 1 to 100
        for i in {1..100}; do
        if ['expr $i % 3' == 0] || [`expr $i % 5' == 0] && [`expr $i % 15 != 0];

        echo $i
