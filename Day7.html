  4. $ bash
  5. $ vim aws_resources_tracker.sh
  6. #!/bin/bash
    ##################
    # Author : mreddy
    # Date: 3-Feb
    # Version : V1
    # This script will report the AWS resource usage
    # AWS S3m AWS EC2, AWS Lambda, AWS IAM
    set -x
    # list s3 buckets
    echo "Print list of s3 buckets" aws s3 ls
    # list ec2 instances
    echo "Print list of ec2 instances"
    aws ec2 describe-instances
    aws ec2 describe-instances | jq '.Reservations[].Instances[].InstanceId' # list lambda functions
    echo "Print list of functions"
    aws lambda list-functions
    # list IAM users echo "Print list of users"
    aws iam list-users
  7. $ chmod 777 aws_resoirces_tracker.sh
